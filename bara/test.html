public static void Main()
{
    // 1. Násobení Matic
    Console.WriteLine("1. Násobení matic:");
    int[,] A = { { 2, 3, 5 }, { 4, 3, 2 }, { 5, 2, 1 }, { 3, 1, 4 } };
    int[,] B = { { 1 }, { 2 }, { 3 } };
    int[,] C = new int[A.GetLength(0), B.GetLength(1)];

    for (int i = 0; i < A.GetLength(0); i++)
    {
        for (int j = 0; j < B.GetLength(1); j++)
        {
            C[i, j] = 0;
            for (int k = 0; k < A.GetLength(1); k++)
            {
                C[i, j] += A[i, k] * B[k, j];
            }
            Console.Write(C[i, j] + " ");
        }
        Console.WriteLine();
    }

    // 2. Výpočet Faktoriálu
    Console.WriteLine("\n2. Výpočet faktoriálu:");
    Console.Write("Zadejte číslo pro výpočet faktoriálu: ");
    int n = int.Parse(Console.ReadLine());
    int faktorial = 1;

    for (int i = 1; i <= n; i++)
    {
        faktorial *= i;
    }
    Console.WriteLine("Faktoriál čísla " + n + " je: " + faktorial);

    // 3. Průměr a Směrodatná Odchylka
    Console.WriteLine("\n3. Průměr a směrodatná odchylka:");
    Random rand = new Random();
    int[] pole = new int[100];
    int soucet = 0;

    for (int i = 0; i < pole.Length; i++)
    {
        pole[i] = rand.Next(1, 101);
        soucet += pole[i];
    }

    double prumer = soucet / (double)pole.Length;
    Console.WriteLine("Průměr: " + prumer);

    double sumaOdchylek = 0;
    for (int i = 0; i < pole.Length; i++)
    {
        sumaOdchylek += Math.Pow(pole[i] - prumer, 2);
    }
    double smerodatnaOdchylka = Math.Sqrt(sumaOdchylek / pole.Length);
    Console.WriteLine("Směrodatná odchylka: " + smerodatnaOdchylka);

    // 4. Vytvoření Pole se Slovem
    Console.WriteLine("\n4. Vytvoření pole se slovem:");
    Console.Write("Zadejte slovo: ");
    string slovo = Console.ReadLine();
    Console.Write("Zadejte počet míst v poli: ");
    int pocet = int.Parse(Console.ReadLine());
    string[] poleSlov = new string[pocet];

    for (int i = 0; i < poleSlov.Length; i++)
    {
        poleSlov[i] = slovo;
        Console.WriteLine("Prvek " + i + ": " + poleSlov[i]);
    }

    // 5. Jednoduchý Kalkulátor
    Console.WriteLine("\n5. Jednoduchý kalkulátor:");
    Console.Write("Zadejte první číslo (X): ");
    double x = double.Parse(Console.ReadLine());
    Console.Write("Zadejte druhé číslo (Y): ");
    double y = double.Parse(Console.ReadLine());
    Console.Write("Zadejte operaci (+, -, *, /): ");
    string operace = Console.ReadLine();

    if (operace == "PLUS")
        Console.WriteLine("Výsledek: " + (x + y));
    else if (operace == "MINUS")
        Console.WriteLine("Výsledek: " + (x - y));
    else if (operace == "KRAT")
        Console.WriteLine("Výsledek: " + (x * y));
    else if (operace == "DELENO")
        Console.WriteLine("Výsledek: " + (x / y));
    else
        Console.WriteLine("Neplatná operace.");
}
